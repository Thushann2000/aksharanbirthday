    <h2 class="gallery-title" data-aos="zoom-in">💖 அக்ஷரனின் அன்பு வாழ்த்துக்கள் 💖</h2>

    <div class="text-center">
        <button id="load-more-btn" class="btn btn-load-more">மேலும் சில வாழ்த்துக்கள் 💝</button>
    </div>

    <script src="https://unpkg.com/aos@2.3.1/dist/aos.js"></script>
    <script>
        AOS.init({ duration: 800, once: true });

        let currentPage = 1;
        const limit = 12; // ஒரு முறைக்கு 12 வாழ்த்துக்கள் மட்டும் லோட் ஆகும்

        async function fetchWishes(page) {
            const container = document.getElementById('wishes-container');
            const loader = document.getElementById('loader');
            const loadMoreBtn = document.getElementById('load-more-btn');
            
            loader.style.display = 'block';
            loadMoreBtn.disabled = true;

            try {
                // Server-இல் நாம் அமைத்த Pagination API-ஐ அழைக்கிறோம்
                const response = await fetch(`/api/signatures?page=${page}&limit=${limit}`);
                const data = await response.json();

                loader.style.display = 'none';
                loadMoreBtn.disabled = false;

                if (data.length < limit) {
                    loadMoreBtn.style.display = 'none'; // மேலும் தரவுகள் இல்லை என்றால் பட்டனை மறைக்கவும்
                }

                data.forEach(item => {
                    const card = document.createElement('div');
                    card.className = 'wish-card';
                    card.setAttribute('data-aos', 'fade-up');
                    
                    card.innerHTML = `
                        <img src="${item.signature}" class="signature-img" alt="Signature" loading="lazy">
                        <span class="signer-name">${item.name || 'அன்புள்ள நண்பர்'}</span>
                        <p class="wish-text">"${item.wish || 'பிறந்தநாள் வாழ்த்துக்கள் குட்டி அக்ஷரன்!'}"</p>
                    `;
                    container.appendChild(card);
                });

                // புதிய கார்டுகளுக்கு AOS அனிமேஷனை புதுப்பிக்கவும்
                AOS.refresh();

            } catch (err) {
                console.error("Error fetching wishes:", err);
                loader.innerText = "மன்னிக்கவும்! தரவுகளைப் பெறுவதில் சிக்கல்.";
            }
        }

        // முதல் பக்கத்தை லோட் செய்யவும்
        fetchWishes(currentPage);

        // Load More பட்டன் செயல்பாடு
        document.getElementById('load-more-btn').addEventListener('click', () => {
            currentPage++;
            fetchWishes(currentPage);
        });
    </script>
