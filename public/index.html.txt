<script>
    AOS.init({ duration: 1000, once: true, offset: 50 });

    // Desktop Parallax
    if (window.innerWidth >= 992) {
        window.addEventListener('scroll', function() {
            const depth = document.querySelectorAll('.parallax');
            for (let i = 0; i < depth.length; i++) {
                const speed = depth[i].getAttribute('data-speed');
                const yPos = -(window.pageYOffset / speed);
                depth[i].style.transform = `translateY(${yPos}px)`;
            }
        });
    }

    const canvas = document.getElementById('signature-pad');
    const signaturePad = new SignaturePad(canvas, { 
        backgroundColor: 'rgb(255,255,255)', 
        penColor: 'rgb(190, 24, 93)',
        minWidth: 2,
        maxWidth: 4
    });

    function resizeCanvas() {
        const ratio = Math.max(window.devicePixelRatio || 1, 1);
        canvas.width = canvas.offsetWidth * ratio;
        canvas.height = canvas.offsetHeight * ratio;
        canvas.getContext("2d").scale(ratio, ratio);
        signaturePad.clear();
    }
    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();

    // Clear Button - மொபைல் வேகத்திற்காக 'touchstart' சேர்க்கப்பட்டது
    const clearBtn = document.getElementById('clear');
    const clearAction = (e) => {
        e.preventDefault();
        signaturePad.clear();
    };
    clearBtn.addEventListener('click', clearAction);
    clearBtn.addEventListener('touchstart', clearAction);

    // Save Button
    const saveBtn = document.getElementById('save');
    const saveAction = async (e) => {
        e.preventDefault();
        if (signaturePad.isEmpty()) { alert("கையெழுத்து போடவும்! 🍼"); return; }
        
        saveBtn.disabled = true;
        saveBtn.innerText = "Saving...";

        // 1. திரையை அதிரச் செய்தல் (NEW: Full Screen Shake)
        document.body.classList.add('shake-active');

        // 2. பட்டாசு வெடிக்கும் எஃபெக்ட் (NEW: Multi-burst Firecrackers)
        const duration = 3 * 1000;
        const animationEnd = Date.now() + duration;
        const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 10000 };

        const randomInRange = (min, max) => Math.random() * (max - min) + min;

        const interval = setInterval(function() {
            const timeLeft = animationEnd - Date.now();
            if (timeLeft <= 0) return clearInterval(interval);

            const particleCount = 50 * (timeLeft / duration);
            confetti({ ...defaults, particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 } });
            confetti({ ...defaults, particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 } });
        }, 250);

        const dataURL = signaturePad.toDataURL("image/jpeg", 0.5); 
        const name = document.getElementById('signer-name').value.trim();
        const wish = document.getElementById('wish').value.trim();
        
        try {
            const response = await fetch('/api/signatures', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ signature: dataURL, name, wish })
            });
            
            if (response.ok) {
                const cake = document.getElementById('celebration-cake');
                cake.classList.add('show');
                
                setTimeout(() => {
                    // அதிரவைப்பதை நிறுத்துதல்
                    document.body.classList.remove('shake-active');
                    cake.classList.remove('show');
                    alert("வாழ்த்துக்கள் சேமிக்கப்பட்டது! ❤️");
                    location.reload();
                }, 3000);
            }
        } catch (err) {
            document.body.classList.remove('shake-active');
            alert("Error! மீண்டும் முயற்சி செய்யவும்.");
            saveBtn.disabled = false;
            saveBtn.innerText = "Save & Celebrate! 🎈";
        }
    };

    saveBtn.addEventListener('click', saveAction);
    saveBtn.addEventListener('touchstart', (e) => {
        if (!saveBtn.disabled) saveAction(e);
    });
</script>
